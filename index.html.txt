    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

        // Ungaludaiya Firebase Configuration
        const firebaseConfig = {
          apiKey: "AIzaSyBJgk2EzELMAeW0dhw2hyTIxtp6f7Tbo8M",
          authDomain: "thivash-32c14.firebaseapp.com",
          projectId: "thivash-32c14",
          storageBucket: "thivash-32c14.firebasestorage.app",
          messagingSenderId: "842996953973",
          appId: "1:842996953973:web:1ebeebc76aa8a2a4ab993c",
          measurementId: "G-FQCXPWXLZ6"
        };

        // Firebase-ah initialize panrom
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);

        let allStyles = [];

        // Database la irunthu data-va eduthu website la kaata
        async function fetchStyles() {
            const querySnapshot = await getDocs(collection(db, "styles"));
            allStyles = [];
            querySnapshot.forEach((doc) => {
                allStyles.push({ id: doc.id, ...doc.data() });
            });
            renderGallery(allStyles);
        }

        // Gallery Grid build panna
        function renderGallery(items) {
            const grid = document.getElementById('galleryGrid');
            grid.innerHTML = '';
            
            if(items.length === 0) {
                grid.innerHTML = '<p style="text-align:center; grid-column: 1/-1;">No styles found!</p>';
                return;
            }

            items.forEach(item => {
                const card = document.createElement('div');
                card.className = 'card';
                card.innerHTML = `
                    <img src="${item.image}" alt="${item.name}" onclick="openModal('${item.id}')">
                    <div class="card-info">
                        <div class="card-title">${item.name}</div>
                        <div class="prompt-box">${item.prompt}</div>
                        <button class="copy-btn" onclick="copyText('${item.prompt.replace(/'/g, "\\'")}')">Copy Prompt</button>
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        // Admin panel Upload logic
        document.getElementById('adminForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            const name = document.getElementById('prodName').value;
            const image = document.getElementById('prodImage').value;
            const prompt = document.getElementById('prodPrompt').value;
            const category = document.getElementById('prodCategory').value;

            try {
                // Firestore "styles" collection-ku upload aahum
                await addDoc(collection(db, "styles"), {
                    name, image, prompt, category, timestamp: new Date()
                });
                alert("Style Successfully Uploaded!");
                document.getElementById('adminForm').reset();
                fetchStyles(); // UI-ah refresh pannum
            } catch (error) {
                console.error("Error adding document: ", error);
            }
        });

        // Search Engine Feature
        document.getElementById('searchInput').addEventListener('input', (e) => {
            const searchKey = e.target.value.toLowerCase();
            const filtered = allStyles.filter(item => 
                item.name.toLowerCase().includes(searchKey) || 
                item.prompt.toLowerCase().includes(searchKey)
            );
            renderGallery(filtered);
        });

        // Category Filter
        window.filterCategory = function(category) {
            // Button Active styling toggle
            document.querySelectorAll('.cat-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');

            if(category === 'All') {
                renderGallery(allStyles);
            } else {
                const filtered = allStyles.filter(item => item.category === category);
                renderGallery(filtered);
            }
        }

        // Global Copy Text Feature
        window.copyText = function(text) {
            navigator.clipboard.writeText(text).then(() => {
                alert("Prompt Copied to Clipboard!");
            });
        }

        // Photo Click panna Detail and More Images open aaharathu
        window.openModal = function(id) {
            const selected = allStyles.find(item => item.id === id);
            if(!selected) return;

            document.getElementById('modalImg').src = selected.image;
            document.getElementById('modalTitle').innerText = selected.name;
            document.getElementById('modalPrompt').innerText = selected.prompt;
            document.getElementById('modalCopyBtn').onclick = () => copyText(selected.prompt);

            // Related/More Images logic (same category-la irukura matha items)
            const moreImages = allStyles.filter(item => item.category === selected.category && item.id !== selected.id);
            const moreGrid = document.getElementById('moreImagesGrid');
            moreGrid.innerHTML = '';

            moreImages.slice(0, 4).forEach(item => {
                const card = document.createElement('div');
                card.className = 'card';
                card.innerHTML = `
                    <img src="${item.image}" alt="${item.name}" onclick="openModal('${item.id}')">
                    <div class="card-info">
                        <div class="card-title">${item.name}</div>
                    </div>
                `;
                moreGrid.appendChild(card);
            });

            document.getElementById('imageModal').style.display = 'block';
        }

        window.closeModal = function() {
            document.getElementById('imageModal').style.display = 'none';
        }

        // Initial Load data fetch
        fetchStyles();
