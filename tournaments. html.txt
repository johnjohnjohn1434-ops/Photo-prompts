    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getAuth, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";
        import { getFirestore, doc, getDoc, collection, addDoc, onSnapshot, deleteDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

        const firebaseConfig = {
            apiKey: "AIzaSyBJgk2EzELMAeW0dhw2hyTIxtp6f7Tbo8M",
            authDomain: "thivash-32c14.firebaseapp.com",
            projectId: "thivash-32c14",
            storageBucket: "thivash-32c14.firebasestorage.app",
            messagingSenderId: "842996953973",
            appId: "1:842996953973:web:1ebeebc76aa8a2a4ab993c",
            measurementId: "G-FQCXPWXLZ6"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);

        let userRole = "player";

        onAuthStateChanged(auth, async (user) => {
            if (!user) {
                window.location.href = "index.html";
            } else {
                const userDoc = await getDoc(doc(db, "users", user.uid));
                if (userDoc.exists()) {
                    userRole = userDoc.data().role || "player";
                }
                if (userRole === 'admin') {
                    document.getElementById('admin-badge').classList.remove('hidden');
                    document.getElementById('admin-add-tn-btn').classList.remove('hidden');
                }
                listenToTournaments();
            }
        });

        function listenToTournaments() {
            onSnapshot(collection(db, "tournaments"), (snapshot) => {
                const container = document.getElementById('tournaments-container');
                const selectDropdown = document.getElementById('reg-tn-id');
                
                container.innerHTML = "";
                selectDropdown.innerHTML = `<option value="" disabled selected>Choose Tournament</option>`;
                
                snapshot.forEach((doc) => {
                    const tn = doc.data();
                    const isAdmin = userRole === 'admin';
                    
                    // Dropdown options filling
                    selectDropdown.innerHTML += `<option value="${tn.title}">${tn.title}</option>`;

                    // Card Generation
                    container.innerHTML += `
                        <div class="gaming-card p-6 rounded-xl flex flex-col md:flex-row justify-between items-start md:items-center gap-4">
                            <div class="space-y-2">
                                <span class="bg-orange-500/10 border border-orange-500/30 text-orange-500 text-[10px] font-black px-2 py-0.5 rounded tracking-widest uppercase">LIVE SLOTS OPEN</span>
                                <h3 class="text-xl font-black text-white italic tracking-wide">${tn.title}</h3>
                                <div class="flex gap-4 text-xs text-gray-400 flex-wrap">
                                    <span class="flex items-center gap-1 text-yellow-500 font-bold"><i data-lucide="gem" class="w-3.5 h-3.5"></i> ${tn.prize}</span>
                                    <span class="flex items-center gap-1"><i data-lucide="calendar" class="w-3.5 h-3.5"></i> ${tn.date}</span>
                                    <span class="flex items-center gap-1"><i data-lucide="users" class="w-3.5 h-3.5"></i> Slots: ${tn.slots}</span>
                                </div>
                            </div>
                            <div class="flex items-center gap-2 w-full md:w-auto justify-end">
                                ${isAdmin ? `<button onclick="deleteTournament('${doc.id}')" class="text-red-500 hover:text-red-400 text-xs font-bold cursor-pointer px-3 py-1 flex items-center gap-1"><i data-lucide="trash-2" class="w-3.5 h-3.5"></i> Close</button>` : ''}
                            </div>
                        </div>
                    `;
                });
                lucide.createIcons();
                if(snapshot.empty) container.innerHTML = `<p class="text-gray-500 text-sm">No active tournaments right now. Stay tuned!</p>`;
            });
        }

        window.adminSaveTournament = async function(e) {
            e.preventDefault();
            await addDoc(collection(db, "tournaments"), {
                title: document.getElementById('new-tn-title').value,
                prize: document.getElementById('new-tn-prize').value,
                date: document.getElementById('new-tn-date').value,
                slots: document.getElementById('new-tn-slots').value,
                createdAt: new Date().toISOString()
            });
            toggleModal('tn-modal');
            e.target.reset();
        }

        window.handleRegisterSquad = async function(e) {
            e.preventDefault();
            await addDoc(collection(db, "registrations"), {
                tournament: document.getElementById('reg-tn-id').value,
                teamName: document.getElementById('reg-team-name').value,
                leaderUID: document.getElementById('reg-leader-uid').value,
                players: document.getElementById('reg-players').value,
                registeredAt: new Date().toISOString()
            });
            e.target.reset();
            alert("🔥 Your Squad has been successfully registered! Get Ready!");
        }

        window.deleteTournament = async function(docId) {
            if(confirm("Are you sure you want to end this tournament?")) {
                await deleteDoc(doc(db, "tournaments", docId));
            }
        }

        window.toggleModal = function(id) {
            document.getElementById(id).classList.toggle('hidden');
        }
    </script>
